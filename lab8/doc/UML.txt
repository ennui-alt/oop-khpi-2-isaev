class CampusServiceRegistry {
    -services: List<CampusService>
    +addService(CampusService): void
    +getServices(): List<CampusService>
    +findById(String): Optional<CampusService>
    +findByStudentName(String): List<CampusService>
}

class CampusServiceFileManager {
    +saveAsText(CampusServiceRegistry, Path): void
    +loadFromText(Path): CampusServiceRegistry
    +serialize(CampusServiceRegistry, Path): void
    +deserialize(Path): CampusServiceRegistry
}

abstract class CampusService {
    -id: String
    -studentName: String
    -date: LocalDate
    -time: LocalTime
    +getType(): String
    +toTextLine(): String
}

class ExamReminder {
    -courseName: String
    -auditorium: String
}

class RoomBooking {
    -roomNumber: String
    -purpose: String
}

class ConsultationSlot {
    -teacherName: String
    -subject: String
}

class InvalidCampusServiceException
class Exception

CampusServiceFileManager --> InvalidCampusServiceException : throws
