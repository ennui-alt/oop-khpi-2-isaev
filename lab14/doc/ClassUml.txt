class Main {
  +main()
}

class Product {
  -id: int
  -name: String
  -description: String
  -price: double
  +hasNormalDescription(): boolean
}

class Order {
  -number: int
  -customerName: String
  -products: List<Product>
  +getTotalAmount(): double
}

class Invoice {
  -orderNumber: int
  -customerName: String
  -totalAmount: double
  -createdAt: LocalDateTime
}

class StoreService {
  +processOrder(): String
  +isDescriptionGood(): boolean
  +createInvoice(): Invoice
}

class OrderProcessingThread {
  -orders: List<Order>
  -processedOrders: List<String>
  +run(): void
}

class ProductDescriptionTask {
  -products: List<Product>
  -productsWithShortDescription: List<Product>
  +run(): void
}

class InvoiceTask {
  -orders: List<Order>
  -invoices: List<Invoice>
  +run(): void
}
