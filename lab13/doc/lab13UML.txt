    -day: WeekDay
    -number: int
    -startTime: LocalTime
    -endTime: LocalTime
    -classroom: String
    -teacher: String
    -subject: String

    +getNumber(): int

}

    MONDAY
    TUESDAY
    WEDNESDAY
    THURSDAY
    FRIDAY
    SATURDAY
    SUNDAY
    +fromText(text: String): WeekDay

    -errors: List<String>

    +getErrors(): List<String>
}
